class Constraint v d where
    {- newConstraint creates a constraint using the given list of variables
    newConstraint :: Ord v => [v] -> c -- -}
    {- satisfied checks for a constraint and
       an assignment = a map of variables each to a value within the corresponding domain,
       if the assignment satisfies the constraint -}
    satisfied :: (Ord v) => v d -> Map.Map v d -> Bool
    {- variables returns the list of variables that are involved in the constraint -}
    constraintVariables :: (Ord v) => v d -> [v]


backtrackingSearch :: (Constraint c, Ord v) => CSP csp v d -> Maybe (Map.Map v d)


class CSP csp where
    {- newCSP creates a new constraint satisfaction problem using
       a mapping of variables to their domains.
       Note that each variable's domain is a list of possible values -}
    newCSP :: Ord v => Map.Map v [d] -> csp
    {- cspVariables get a CSP's list of variables -}
    cspVariables :: CSP csp => csp -> [v]
    {- domains get a CSP's domains for each variable -}
    domains :: (CSP csp, Ord v) => csp -> Map.Map v [d]
    {- addConstraint "adds" a constraint to a CSP -}
    addConstraint :: (CSP csp, Constraint c) => csp -> c -> csp
    {-isConsistent , for a given CSP, if the given
      assignment = a map of variables each to a value within the corresponding domain,
      is consistent with the CSP -}
    isConsistent :: CSP csp => csp -> v -> Map.Map v d -> Bool
    {- backtracking_search performs a backtracking search for a given CSP
       to find a solution = assignment for each variable of the CSP to a value within the corresponding domain
       that satisfies all constraints of the given CSP. -}
    backtrackingSearch :: (CSP csp, Ord v) => csp -> Maybe (Map.Map v d)
    backtrackingSearch csp = internalBacktrackingSearch csp Map.empty  -- :: CSP csp => csp -> Maybe (Map.Map v d)

internalBacktrackingSearch :: (CSP csp, Ord v) => csp -> Map.Map v d -> Maybe (Map.Map v d)
internalBacktrackingSearch csp assignment
    | length assignment == length (cspVariables csp) = Just assignment -- assignment is complete if every variable is assigned (our base case)
    | otherwise = do
        let unassigned = filter (`Map.notMember` assignment) (cspVariables csp)
        let (firstUnassigned:xs) = unassigned -- get first unassigned variable -> next its possible values
        let domain = case Map.lookup firstUnassigned (domains csp) of
                        Just d -> d
                        Nothing -> []
        tryValues csp firstUnassigned domain assignment

tryValues csp firstUnassigned [] assignment = Nothing
tryValues csp firstUnassigned (value:values) assignment
    | isConsistent csp firstUnassigned (Map.insert firstUnassigned value assignment) =
        case internalBacktrackingSearch csp (Map.insert firstUnassigned value assignment) of
            Just result -> Just result
            Nothing -> tryValues csp firstUnassigned values assignment
    | otherwise = tryValues csp firstUnassigned values assignment
